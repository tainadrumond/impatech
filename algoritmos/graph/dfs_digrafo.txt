# DFS PARA GRAFOS DIRECIONADOS

t = 0

for v in V(G):
    pai(v) = null
    PE(v) = 0
    PS(v) = 0

while existe v em V(G) tal que PE(v) == 0:
    DFS(v)

function DFS(v):
    t = t+1
    PE(v) = t

    for w in N_out(v):
        if PE(w) == 0:
            visitar vw // aresta azul de profundidade 
            pai(w) = v
            DFS(w)
        
        else if PS(w) == 0:
            visitar vw // aresta vermelha de retorno
        
        else if PE(w) > PE(v):
            visitar vw // aresta amarela de avanço
        
        else:
            visitar vw // aresta verde de cruzamento

    t = t+1
    PS(v) = t
